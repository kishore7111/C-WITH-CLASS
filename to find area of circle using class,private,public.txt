#include<iostream>
using namespace std;
class AREA
{
    private:
        double radius;
    public:
        void area(double r)
        {
            double area;
            radius = r;
            area = 3.14*radius*radius;
            cout << "Radius is:" << radius << endl;
            cout << "Area is:" << area;
        }
};
int main()
{
    AREA obj;
    obj.area(5);
}
